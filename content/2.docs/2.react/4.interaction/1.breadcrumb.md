```jsx [DemoBreadcrumbBase.tsx]
import { BaseBreadcrumb } from "@shuriken-ui/react";

const breadcrumb = [
  {
    label: 'Home',
    hideLabel: false,
    icon: 'lucide:home',
    to: '#',
  },
  {
    label: 'Products',
    hideLabel: false,
    to: '#',
  },
  {
    label: 'Laptops',
    hideLabel: false,
    to: '#',
  },
]

export default function Page() {
  return (
    <div>
      <BaseBreadcrumb items={breadcrumb} separator="dot" />
    </div>
  )
}

```

  {
    label: 'Home',
    hideLabel: false,
    icon: 'lucide:home',
    to: '#',
