### Action

```tsx [DemoInputAction.tsx]

  const [value, setValue] = useState<string>();
  return (
    <div>
      <BaseInput
        value={value}
        onChange={(value) => setValue(value)}
        label="Username"
        placeholder="Ex: username"
        action={
          <button
            type="button"

}
```
